automacro AutoLootOnConnect {
    OnCharLogIn 1
    timeout 3
    call {
        do c \@autoloot 0.5
        log Autoloot activated
    }
}

automacro sp_exchange {
	PubMsgDist /sp me/ < 9
	call {
		do sp 374 $.PubMsgDistLastID
	}
}
automacro Wait {
	CurrentHP < 50%
	delay 100
	run-once 1
    call {
		do conf route_randomWalk 0
        log Followed due to min HP
    }
}
automacro Wait1 {
	CurrentHP > 70%
	run-once 1
    call {
        do conf route_randomWalk 1
        log follow ok
    }
}
